## Logistic回归与敏感性分析
library(data.table)
library(dplyr)
library(logistf)
#library(MatchIt)

phenotype <- fread("/home1/wangliang/MS_proteomic/data_MS.csv")
covariates <- fread("/home1/wangliang/IMD/covarites_imputed0118.csv")
proteomics <- fread("/home1/wangliang/IMD/ProteinData_new.csv")
proteomics_covariates <- fread("/home1/wangliang/IMD/ProteinSuppInfo_imputed1228.csv")
phenotype1 <- subset(phenotype, !(status == 1 & disease_years > 0))

data_all <- phenotype1 %>%
  inner_join(covariates, by = "eid") %>%
  inner_join(proteomics, by = "eid") %>%
  inner_join(proteomics_covariates, by = "eid")

#data_all1 <- matchit(status~ age+sex,
                  method = "nearest", 
                  distance = "mahalanobis" , 
                  ratio = 5/10, 
                  link = "logit" , 
                  exact = ~sex,  
                  data = data_all)
#data_all1 <- match.data(data_all1)

# 第一阶段：运行普通glm并记录警告
results <- data.frame()
protein_names <- names(proteomics)[-1]
warnings_list <- list()  # 存储警告信息
needs_redo <- character()  # 需要重新分析的蛋白质

for (protein in protein_names) {
  cat("第一阶段 - 正在分析:", protein, "\n")
  
  # 1. 数据准备
  tmp_data <- data_all %>%
    select(eid, status, age, sex, fastingtime, season, 
           smoke, TDI, BMI, ethnicity, alcohol,  
           !!sym(protein), !!sym(paste0(protein, "_SampAge"))) %>%
    rename(
      x_pro = !!sym(protein),
      x_pro_sa = !!sym(paste0(protein, "_SampAge"))
    ) %>%
    filter(!is.na(x_pro))  # 删除蛋白质值缺失的样本
  
  rownames(tmp_data) <- NULL
  
  # 2. 处理分类变量
  tmp_data <- tmp_data %>%
    mutate(
      sex = as.factor(sex),
      ethnicity = as.factor(ethnicity),
      season = as.factor(season),
      smoke = as.factor(smoke),
      across(c(age, fastingtime, TDI, BMI, alcohol, x_pro, x_pro_sa), as.numeric)
    )
  
  # 3. 移除缺失值
  tmp_data <- tmp_data[complete.cases(tmp_data), ]
  
  # 4. 计算基本统计量
  nb_all <- nrow(tmp_data)
  nb_case <- sum(tmp_data$status, na.rm = TRUE)
  prop_case <- ifelse(nb_all > 0, round(nb_case / nb_all * 100, 3), 0)
  
  # 检查是否有足够的事件
  if (nb_all == 0 || nb_case < 10) {
    cat(sprintf("  跳过: 样本不足 (总数=%d, 事件=%d)\n", nb_all, nb_case))
    
    tmpout <- list(
      protein = protein,
      method = "skipped",
      nb_all = nb_all,
      nb_case = nb_case,
      prop_case = prop_case,
      coef = NA,
      OR = NA,
      lower_95_CI = NA,
      upper_95_CI = NA,
      Pvalue = NA,
      warning = ifelse(nb_all == 0, "No samples", 
                       ifelse(nb_case < 10, "Too few events (<10)", NA))
    )
    
    results <- bind_rows(results, as.data.frame(tmpout))
    next
  }
  
  # 5. 创建Logistic回归公式
  formula <- as.formula(paste("status ~ x_pro + age + sex + fastingtime + season +",
                              "smoke + TDI + BMI + ethnicity + alcohol + x_pro_sa"))
  
  # 初始化tmpout
  tmpout <- NULL
  
  # 6. 运行glm并捕获警告
  tryCatch({
    warning_messages <- character()
    
    fit <- withCallingHandlers(
      {
        glm(formula, data = tmp_data, family = binomial(link = "logit"))
      },
      warning = function(w) {
        warning_messages <<- c(warning_messages, w$message)
        invokeRestart("muffleWarning")
      }
    )
    
    # 保存警告信息
    if (length(warning_messages) > 0) {
      warning_text <- paste(warning_messages, collapse = "; ")
      warnings_list[[protein]] <- warning_text
      
      # 检查拟合问题警告
      if (grepl("fitted probabilities numerically 0 or 1 occurred|algorithm did not converge", 
                warning_text, ignore.case = TRUE)) {
        cat("  检测到完全分离或拟合问题，标记需要重新分析\n")
        needs_redo <- c(needs_redo, protein)
        method_used <- "glm(with_warning)"
      } else {
        method_used <- "glm"
      }
    } else {
      warnings_list[[protein]] <- NA
      method_used <- "glm"
    }
    
    # 从glm提取结果
    summary_fit <- summary(fit)
    coef_summary <- as.data.frame(summary_fit$coefficients)
    
    if ("x_pro" %in% rownames(coef_summary)) {
      coef_val <- coef_summary["x_pro", "Estimate"]
      OR <- exp(coef_val)
      p_value <- coef_summary["x_pro", "Pr(>|z|)"]
      
      # 获取置信区间
      ci_matrix <- tryCatch({
        confint(fit)
      }, error = function(e) {
        NULL
      })
      
      if (!is.null(ci_matrix) && "x_pro" %in% rownames(ci_matrix)) {
        ci_lower <- exp(ci_matrix["x_pro", 1])
        ci_upper <- exp(ci_matrix["x_pro", 2])
      } else {
        # 备选方法：使用标准误计算
        se <- coef_summary["x_pro", "Std. Error"]
        ci_lower <- exp(coef_val - 1.96 * se)
        ci_upper <- exp(coef_val + 1.96 * se)
      }
      
      # 检查系数是否异常
      if (abs(coef_val) > 10 || is.na(ci_lower) || is.na(ci_upper) || 
          ci_upper > 1e10 || ci_lower < 1e-10) {
        if (!protein %in% needs_redo) {
          cat("  检测到异常系数，标记需要重新分析\n")
          needs_redo <- c(needs_redo, protein)
        }
      }
      
      tmpout <- list(
        protein = protein,
        method = method_used,
        nb_all = nb_all,
        nb_case = nb_case,
        prop_case = prop_case,
        coef = round(coef_val, 5),
        OR = round(OR, 5),
        lower_95_CI = round(ci_lower, 5),
        upper_95_CI = round(ci_upper, 5),
        Pvalue = ifelse(p_value < 0.00001, 
                        as.numeric(format(p_value, scientific = TRUE, digits = 4)), 
                        round(p_value, 5)),
        warning = ifelse(length(warning_messages) > 0, 
                         warning_text, 
                         NA_character_),
        needs_redo = protein %in% needs_redo
      )
      
      cat(sprintf("  使用%s成功拟合: OR=%.3f (%.3f-%.3f), p=%s\n", 
                  method_used, OR, ci_lower, ci_upper, 
                  ifelse(p_value < 0.001, format(p_value, scientific = TRUE), 
                         round(p_value, 4))))
      
    } else {
      stop("x_pro不在结果中")
    }
    
  }, error = function(e) {
    error_msg <- e$message
    cat(sprintf("  分析失败: %s\n", error_msg))
    
    tmpout <<- list(
      protein = protein,
      method = "glm_failed",
      nb_all = nb_all,
      nb_case = nb_case,
      prop_case = prop_case,
      coef = NA,
      OR = NA,
      lower_95_CI = NA,
      upper_95_CI = NA,
      Pvalue = NA,
      warning = error_msg,
      needs_redo = FALSE
    )
  })
  
  # 7. 添加到结果数据框
  if (!is.null(tmpout)) {
    results <- bind_rows(results, as.data.frame(tmpout))
  } else {
    tmpout_default <- list(
      protein = protein,
      method = "unknown_error",
      nb_all = nb_all,
      nb_case = nb_case,
      prop_case = prop_case,
      coef = NA,
      OR = NA,
      lower_95_CI = NA,
      upper_95_CI = NA,
      Pvalue = NA,
      warning = "Analysis failed without error message",
      needs_redo = FALSE
    )
    results <- bind_rows(results, as.data.frame(tmpout_default))
  }
  
  # 每分析10个蛋白质输出一次进度
  if (which(protein == protein_names) %% 10 == 0) {
    cat(sprintf("第一阶段进度: %d/%d 个蛋白质\n", 
                which(protein == protein_names), length(protein_names)))
  }
}

cat(sprintf("\n第一阶段完成！需要重新分析的蛋白质数量: %d\n", length(needs_redo)))
if (length(needs_redo) > 0) {
  cat("需要重新分析的蛋白质:\n")
  print(needs_redo)
  
  # 第二阶段：对有问题的蛋白质使用Firth's logistic回归
  cat("\n开始第二阶段：对有问题的蛋白质使用Firth's logistic回归\n")
  
  # 创建第二阶段结果副本
  results_phase2 <- results
  
  for (i in seq_along(needs_redo)) {
    protein <- needs_redo[i]
    cat(sprintf("\n第二阶段 - 重新分析 %d/%d: %s\n", i, length(needs_redo), protein))
    
    # 获取该蛋白质的原始数据
    tmp_data <- data_all %>%
      select(eid, status, age, sex, fastingtime, season, 
             smoke, TDI, BMI, ethnicity, alcohol,  
             !!sym(protein), !!sym(paste0(protein, "_SampAge"))) %>%
      rename(
        x_pro = !!sym(protein),
        x_pro_sa = !!sym(paste0(protein, "_SampAge"))
      ) %>%
      filter(!is.na(x_pro))
    
    rownames(tmp_data) <- NULL
    
    # 处理分类变量
    tmp_data <- tmp_data %>%
      mutate(
        sex = as.factor(sex),
        ethnicity = as.factor(ethnicity),
        season = as.factor(season),
        smoke = as.factor(smoke),
        across(c(age, fastingtime, TDI, BMI, alcohol, x_pro, x_pro_sa), as.numeric)
      )
    
    # 移除缺失值
    tmp_data <- tmp_data[complete.cases(tmp_data), ]
    
    # 创建公式
    formula <- as.formula(paste("status ~ x_pro + age + sex + fastingtime + season +",
                                "smoke + TDI + BMI + ethnicity + alcohol + x_pro_sa"))
    
    # 使用logistf进行Firth's回归
    logistf_success <- FALSE
    
    tryCatch({
      # 尝试不同的maxiter值
      maxiter_values <- c(50, 100, 200, 500, 1000)
      
      for (maxiter in maxiter_values) {
        tryCatch({
          fit <- logistf(formula, data = tmp_data, maxiter = maxiter)
      
      # 从logistf提取结果
      summary_fit <- summary(fit)
      coef_summary <- as.data.frame(summary_fit$coefficients)
      
      if ("x_pro" %in% rownames(coef_summary)) {
        coef_val <- coef_summary["x_pro", "coef"]
        OR <- exp(coef_val)
        p_value <- coef_summary["x_pro", "p"]
        
        # 获取置信区间
        ci_matrix <- tryCatch({
          confint(fit)
        }, error = function(e) {
          NULL
        })
        
        if (!is.null(ci_matrix) && "x_pro" %in% rownames(ci_matrix)) {
          ci_lower <- exp(ci_matrix["x_pro", 1])
          ci_upper <- exp(ci_matrix["x_pro", 2])
        } else {
          se <- coef_summary["x_pro", "se(coef)"]
          ci_lower <- exp(coef_val - 1.96 * se)
          ci_upper <- exp(coef_val + 1.96 * se)
        }
        
        # 找到该蛋白质在结果中的位置
        protein_idx <- which(results_phase2$protein == protein)
        
        if (length(protein_idx) > 0) {
          # 更新结果
          results_phase2$method[protein_idx] <- "logistf(Firth)"
          results_phase2$coef[protein_idx] <- round(coef_val, 5)
          results_phase2$OR[protein_idx] <- round(OR, 5)
          results_phase2$lower_95_CI[protein_idx] <- round(ci_lower, 5)
          results_phase2$upper_95_CI[protein_idx] <- round(ci_upper, 5)
          results_phase2$Pvalue[protein_idx] <- ifelse(p_value < 0.00001, 
                                                       format(p_value, scientific = TRUE, digits = 4), 
                                                       round(p_value, 5))
          
          logistf_success <- TRUE
          cat(sprintf("  使用logistf(Firth)成功: OR=%.3f (%.3f-%.3f), p=%s\n", 
                      OR, ci_lower, ci_upper, 
                      ifelse(p_value < 0.001, format(p_value, scientific = TRUE), 
                             round(p_value, 4))))
          break
        }
      }
        }, error = function(e) {
          # 继续尝试下一个maxiter
        })
      }
      
      if (!logistf_success) {
        cat("  警告: logistf分析失败，保留原始glm结果\n")
      }
      
    }, error = function(e) {
      cat(sprintf("  重新分析失败: %s\n", e$message))
    })
  }
  
  # 更新最终结果
  results <- results_phase2
}

# 清理和总结
cat("\n=== 分析完成！ ===\n")

# 移除临时列
if ("needs_redo" %in% names(results)) {
  results <- results %>% select(-needs_redo)
}

  }

      # 检查是否有无限系数警告
      if (grepl("Loglik converged before variable|coefficient may be infinite", 
                warning_text, ignore.case = TRUE)) {
        cat("  检测到无限系数警告，标记需要重新分析\n")
        needs_redo <- c(needs_redo, protein)
        method_used <- "coxph(with_warning)"
      } else {
        method_used <- "coxph"
      }
    } else {
      warnings_list[[protein]] <- NA
      method_used <- "coxph"
    }
    
    # 从coxph提取结果
    summary_fit <- summary(fit)
    coef_summary <- as.data.frame(summary_fit$coefficients)
    
    if ("x_pro" %in% rownames(coef_summary)) {
      coef_val <- coef_summary["x_pro", "coef"]
      HR <- summary_fit$conf.int["x_pro", "exp(coef)"]
      p_value <- coef_summary["x_pro", "Pr(>|z|)"]
      
      # 获取置信区间
      ci_lower <- summary_fit$conf.int["x_pro", "lower .95"]
      ci_upper <- summary_fit$conf.int["x_pro", "upper .95"]
      
      # 也检查系数是否异常（即使没有警告）
      if (abs(coef_val) > 10 || is.na(ci_lower) || is.na(ci_upper) || 
          ci_upper > 1e10 || ci_lower < 1e-10) {
        if (!protein %in% needs_redo) {
          cat("  检测到异常系数，标记需要重新分析\n")
          needs_redo <- c(needs_redo, protein)
        }
      }
      
      tmpout <- list(
        protein = protein,
        method = method_used,
        nb_all = nb_all,
        nb_case = nb_case,
        prop_case = prop_case,
        coef = round(coef_val, 5),
        HR = round(HR, 5),
        lower_95_CI = round(ci_lower, 5),
        upper_95_CI = round(ci_upper, 5),
        Pvalue = ifelse(p_value < 0.00001, 
                        as.numeric(format(p_value, scientific = TRUE, digits = 4)), 
                        round(p_value, 5)),
        warning = ifelse(length(warning_messages) > 0, 
                         warning_text, 
                         NA_character_),
        needs_redo = protein %in% needs_redo
      )
      
      cat(sprintf("  使用%s成功拟合: HR=%.3f (%.3f-%.3f), p=%s\n", 
                  method_used, HR, ci_lower, ci_upper, 
                  ifelse(p_value < 0.001, format(p_value, scientific = TRUE), 
                         round(p_value, 4))))
      
    } else {
      stop("x_pro不在结果中")
    }
    
  }, error = function(e) {
    error_msg <- e$message
    cat(sprintf("  分析失败: %s\n", error_msg))
    
    tmpout <<- list(
      protein = protein,
      method = "coxph_failed",
      nb_all = nb_all,
      nb_case = nb_case,
      prop_case = prop_case,
      coef = NA,
      HR = NA,
      lower_95_CI = NA,
      upper_95_CI = NA,
      Pvalue = NA,
      warning = error_msg,
      needs_redo = FALSE
    )
  })
  
  # 7. 添加到结果数据框
  if (!is.null(tmpout)) {
    results <- bind_rows(results, as.data.frame(tmpout))
  } else {
    # 如果tmpout仍然为NULL，创建一个默认的失败记录
    tmpout_default <- list(
      protein = protein,
      method = "unknown_error",
      nb_all = nb_all,
      nb_case = nb_case,
      prop_case = prop_case,
      coef = NA,
      HR = NA,
      lower_95_CI = NA,
      upper_95_CI = NA,
      Pvalue = NA,
      warning = "Analysis failed without error message",
      needs_redo = FALSE
    )
    results <- bind_rows(results, as.data.frame(tmpout_default))
  }
  
  # 每分析10个蛋白质输出一次进度
  if (which(protein == protein_names) %% 10 == 0) {
    cat(sprintf("第一阶段进度: %d/%d 个蛋白质\n", 
                which(protein == protein_names), length(protein_names)))
  }
}

cat(sprintf("\n第一阶段完成！需要重新分析的蛋白质数量: %d\n", length(needs_redo)))
if (length(needs_redo) > 0) {
  cat("需要重新分析的蛋白质:\n")
  print(needs_redo)
  
  # 第二阶段：对有警告的蛋白质重新用coxphf分析
  cat("\n开始第二阶段：对有警告的蛋白质使用coxphf重新分析\n")
  
  # 创建第二阶段结果副本
  results_phase2 <- results
  
  for (i in seq_along(needs_redo)) {
    protein <- needs_redo[i]
    cat(sprintf("\n第二阶段 - 重新分析 %d/%d: %s\n", i, length(needs_redo), protein))
    
    # 获取该蛋白质的原始数据
    tmp_data <- data_all %>%
      select(eid, disease_days, status, age, sex, fastingtime, season, 
             smoke, TDI, BMI, ethnicity, alcohol,  
             !!sym(protein), !!sym(paste0(protein, "_SampAge"))) %>%
      rename(
        x_pro = !!sym(protein),
        x_pro_sa = !!sym(paste0(protein, "_SampAge"))
      ) %>%
      filter(!is.na(x_pro))
    
    rownames(tmp_data) <- NULL
    
    # 处理分类变量
    tmp_data <- tmp_data %>%
      mutate(
        sex = as.factor(sex),
        ethnicity = as.factor(ethnicity),
        season = as.factor(season),
        smoke = as.factor(smoke),
        across(c(age, fastingtime, TDI, BMI, alcohol, x_pro, x_pro_sa), as.numeric)
      )
    
    # 移除缺失值
    tmp_data <- tmp_data[complete.cases(tmp_data), ]
    
    # 创建生存对象和公式
    surv_obj <- Surv(tmp_data$disease_days, tmp_data$status)
    formula <- as.formula(paste("surv_obj ~ x_pro + age + sex + fastingtime + season +",
                                "smoke + TDI + BMI + ethnicity + alcohol + x_pro_sa"))
    
    # 使用coxphf重新分析
    coxphf_success <- FALSE
    
    tryCatch({
      # 尝试不同的maxiter值
      maxiter_values <- c(50, 100, 200, 500, 1000)
      
      for (maxiter in maxiter_values) {
        tryCatch({
          fit <- coxphf(formula, data = tmp_data, maxiter = maxiter)
          
          # 从coxphf提取结果
          summary_fit <- summary(fit)
          coef_summary <- as.data.frame(summary_fit$coefficients)
          
          if ("x_pro" %in% rownames(coef_summary)) {
            coef_val <- coef_summary["x_pro", "coef"]
            HR <- exp(coef_val)
            p_value <- coef_summary["x_pro", "prob"]
            
            # 获取置信区间
            ci_matrix <- tryCatch({
              confint(fit)
            }, error = function(e) {
              NULL
            })
            
            if (!is.null(ci_matrix) && "x_pro" %in% rownames(ci_matrix)) {
              ci_lower <- exp(ci_matrix["x_pro", 1])
              ci_upper <- exp(ci_matrix["x_pro", 2])
            } else {
              se <- coef_summary["x_pro", "se(coef)"]
              ci_lower <- exp(coef_val - 1.96 * se)
              ci_upper <- exp(coef_val + 1.96 * se)
            }
            
            # 找到该蛋白质在结果中的位置
            protein_idx <- which(results_phase2$protein == protein)
            
            if (length(protein_idx) > 0) {
              # 更新结果
              results_phase2$method[protein_idx] <- paste0("coxphf(maxiter=", maxiter, ")")
              results_phase2$coef[protein_idx] <- round(coef_val, 5)
              results_phase2$HR[protein_idx] <- round(HR, 5)
              results_phase2$lower_95_CI[protein_idx] <- round(ci_lower, 5)
              results_phase2$upper_95_CI[protein_idx] <- round(ci_upper, 5)
              results_phase2$Pvalue[protein_idx] <- ifelse(p_value < 0.00001, 
                                                           format(p_value, scientific = TRUE, digits = 4), 
                                                           round(p_value, 5))
              
              coxphf_success <- TRUE
              cat(sprintf("  使用coxphf(maxiter=%d)成功: HR=%.3f (%.3f-%.3f), p=%s\n", 
                          maxiter, HR, ci_lower, ci_upper, 
                          ifelse(p_value < 0.001, format(p_value, scientific = TRUE), 
                                 round(p_value, 4))))
              break
            }
          }
        }, error = function(e) {
          # 继续尝试下一个maxiter
        })
      }
      
      if (!coxphf_success) {
        cat("  警告: coxphf所有尝试均失败，保留原始coxph结果\n")
      }
      
    }, error = function(e) {
      cat(sprintf("  重新分析失败: %s\n", e$message))
    })
  }
  
  # 更新最终结果
  results <- results_phase2
}

def calculate_specificity(y_test, y_pred):
     """计算特异性"""
     tn, fp, fn, tp = confusion_matrix(y_test, y_pred).ravel()
     return tn / (tn + fp) if (tn + fp) > 0 else 0 
def calculate_youden(y_test, y_pred):
     """计算Youden指数"""
     tn, fp, fn, tp = confusion_matrix(y_test, y_pred).ravel()
     sensitivity = tp / (tp + fn) if (tp + fn) > 0 else 0
     specificity = tn / (tn + fp) if (tn + fp) > 0 else 0
     return sensitivity + specificity - 1

def calculate_ci(scores, alpha=0.05):
        """计算置信区间"""
        lower = np.percentile(scores, alpha/2 * 100)
        upper = np.percentile(scores, (1 - alpha/2) * 100)
        mean = np.mean(scores)
        return mean, (lower, upper)

from sklearn.model_selection import train_test_split
import lightgbm as lgb
import warnings
warnings.filterwarnings('ignore')

X = pd.read_csv('/home1/wangliang/IMD/ProteinData_new.csv')
Y = pd.read_csv('/home1/wangliang/MS_proteomic/data_MS.csv')
Y1 = Y[~( (Y['status'] == 1) & (Y['disease_years'] > 0) )]
Z = pd.read_csv('/home1/wangliang/IMD/covarites_imputed0118.csv')
EX = pd.read_csv('/home1/wangliang/MS_proteomic/exvalidation1.csv')
Y2 = pd.merge(X, Y1, on='eid')
y = pd.merge(Y2, Z, on='eid')
df1 = pd.DataFrame(y)
y = df1[['status']]
x = df1[['NEFL', 'GFAP', 'TF', 'ITGAV', 'GZMA', 'CST1', 'ADGRG2', 'CA6', 'ACP5', 
        'CLEC4A', 'BCAN', 'OMG', 'CST5', 'HPGDS', 'MSTN', 'ITGB6', 'FGFBP2', 'COL15A1', 
        'KLK6', 'AMY1A_AMY1B_AMY1C', 'MERTK', 'CPA2', 'ITGAM', 'SFRP4', 'CRTAC1',   
        'age', 'sex', 'ethnicity', 'alcohol', 'smoke', 'BMI']]
df2 = pd.DataFrame(EX)
y_test = df2[['status']]
x_test = df2[['NEFL', 'GFAP', 'CA6', 'BCAN', 'CST5', 'HPGDS', 'KLK6', 
         'MERTK', 'CPA2', 'ITGAM', 'age', 'sex', 'ethnicity']]

# 处理分类变量
categorical_features_indices = []
# 确保数据类型合适
for i, col in enumerate(x.columns):
    if col in ['sex', 'ethnicity', 'smoke']:  # 或者检查数据类型
        categorical_features_indices.append(i)

x_train, x_val, y_train, y_val=train_test_split(x, y, random_state=42, test_size=0.3)
model = lgb.LGBMClassifier(objective='binary', metric='auc', is_unbalance=True, verbosity= 1, 
                           seed=2023)
model.set_params(**{'n_estimators': 500, 'max_depth': 15, 'num_leaves': 10,
                             'subsample': 0.7, 'learning_rate': 0.01, 'colsample_bytree': 0.7})
model.fit(x_train, y_train, categorical_feature=categorical_features_indices)

y_pred_prob = model.predict_proba(x_test, predict_disable_shape_check=True)[:, 1]
y_test = y_test.values.ravel()
threshold = 0.5
y_pred = (y_pred_prob >= threshold).astype(int)
metrics = {}
data = pd.DataFrame({
           'true': y_test,
           'pred_proba': y_pred_prob,
           'pred': y_pred
       })
def calculate_specificity(y_test, y_pred):
     """计算特异性"""
     tn, fp, fn, tp = confusion_matrix(y_test, y_pred).ravel()
     return tn / (tn + fp) if (tn + fp) > 0 else 0 
def calculate_youden(y_test, y_pred):
     """计算Youden指数"""
     tn, fp, fn, tp = confusion_matrix(y_test, y_pred).ravel()
     sensitivity = tp / (tp + fn) if (tp + fn) > 0 else 0
     specificity = tn / (tn + fp) if (tn + fp) > 0 else 0
     return sensitivity + specificity - 1
n_bootstraps=1000
random_state=42
np.random.seed(random_state)
auc_scores = []
accuracy_scores = []
recall_scores = []
specificity_scores = []
precision_scores = []
youden_scores = []
f1_scores = []
# Bootstrap抽样
for i in tqdm(range(n_bootstraps), desc="Bootstrap采样"):
        # 有放回抽样
        sample = resample(data, replace=True, random_state=random_state+i)
        
        # 计算各项指标
        try:
            # AUC
            auc = roc_auc_score(sample['true'], sample['pred_proba'])
            auc_scores.append(auc)
            
            # 其他基于阈值的指标
            acc = accuracy_score(sample['true'], sample['pred'])
            rec = recall_score(sample['true'], sample['pred'], zero_division=0)
            spec = calculate_specificity(sample['true'], sample['pred'])
            prec = precision_score(sample['true'], sample['pred'], zero_division=0)
            yden = calculate_youden(sample['true'], sample['pred'])
            f1 = f1_score(sample['true'], sample['pred'], zero_division=0)
            
            accuracy_scores.append(acc)
            recall_scores.append(rec)
            specificity_scores.append(spec)
            precision_scores.append(prec)
            youden_scores.append(yden)
            f1_scores.append(f1)
            
        except:
            continue
def calculate_ci(scores, alpha=0.05):
        """计算置信区间"""
        lower = np.percentile(scores, alpha/2 * 100)
        upper = np.percentile(scores, (1 - alpha/2) * 100)
        mean = np.mean(scores)
        return mean, (lower, upper)
results = {}
metrics_data = {
        'AUC': auc_scores,
        'Accuracy': accuracy_scores,
        'Recall': recall_scores,
        'Specificity': specificity_scores,
        'Precision': precision_scores,
        'Youden': youden_scores,
        'F1': f1_scores
    }
for metric_name, scores in metrics_data.items():
        mean, ci = calculate_ci(scores)
        results[metric_name] = {
            'mean': mean,
            '95%_CI': ci,
            'CI_lower': ci[0],
            'CI_upper': ci[1]
        }
def print_results(results):
    """格式化打印结果"""
    print("Bootstrap结果 (95% 置信区间):")
    print("=" * 60)
    for metric, values in results.items():
        print(f"{metric:12s}: {values['mean']:.4f} ({values['CI_lower']:.4f} - {values['CI_upper']:.4f})")
print_results(results)
